How to Use These Videos

To use these videos in your website, copy the URL under each video and use it in your HTML like this:

<video autoplay muted loop playsinline>
    <source src="https://dantemagg.github.io/innosolve-assets/videos/FILENAME.mp4" type="video/mp4">
</video>

For hover effects, you can use CSS:

<div class="video-container">
    <video 
        src="https://dantemagg.github.io/innosolve-assets/videos/FILENAME.mp4" 
        class="hover-video"
        muted
        loop
        playsinline></video>
</div>

<style>
.video-container {
    position: relative;
    overflow: hidden;
    border-radius: 8px;
}
.hover-video {
    width: 100%;
    transition: transform 0.3s ease;
}
.video-container:hover .hover-video {
    transform: scale(1.1);
}
.video-container:hover video {
    /* This will autoplay the video on hover */
}
</style>

<script>
// For videos that autoplay on hover
document.querySelectorAll('.video-container').forEach(container => {
    const video = container.querySelector('video');
    
    container.addEventListener('mouseenter', () => {
        video.play();
    });
    
    container.addEventListener('mouseleave', () => {
        video.pause();
        // Optional: reset to first frame
        video.currentTime = 0;
    });
});
</script>
